```python
def generate_template_description(business: dict) -> str:
    """Generate a 2–3 sentence description from structured business data.
    Never call this if the record already has a description — check before calling.
    """
    name     = business.get('name') or business.get('business_name') or 'This business'
    category = business.get('category') or 'business'
    area     = business.get('area') or business.get('location') or 'the Vaal Triangle'

    sentence1 = (
        f"{name} is a {category.lower()} business located in {area}, "
        f"serving the Vaal Triangle community."
    )
    parts = [sentence1]

    rating  = business.get('google_rating')
    reviews = business.get('google_review_count')
    if rating:
        rating_str = f"Rated {rating}\u2605 on Google"
        if reviews:
            rating_str += f" with {int(reviews)} reviews"
        parts.append(rating_str + ".")

    tags_raw = business.get('tags') or ''
    if tags_raw:
        tag_list = [t.strip().lower() for t in tags_raw.split(',') if t.strip()][:3]
        if tag_list:
            parts.append(f"Services include {', '.join(tag_list)}.")

    return ' '.join(parts)
```

```python
    def _run_bulk_publish(self):
        """Publish all pending businesses that have phone + address in a Vaal area.
        Generates template descriptions for blank ones first, then bulk-updates status.
        Runs in a background thread to keep the UI responsive.
        """
        import threading
        threading.Thread(target=self._run_bulk_publish_worker, daemon=True).start()

    def _run_bulk_publish_worker(self):
        import sqlite3 as _sqlite3
        from database import generate_template_description

        VAAL_AREAS = (
            'Vereeniging', 'Vanderbijlpark', 'Meyerton', 'Sasolburg', 'Sebokeng',
            'Evaton', 'Heidelberg', 'Walkerville', 'Three Rivers', 'Randvaal',
            'Orange Farm', 'Vaalpark', 'Henley On Klip', 'De Deur', 'Ennerdale',
            'Sharpeville', 'Duncanville', 'Risiville', 'Stretford', 'Sonlandpark',
            'Delmore', 'Arcon Park', 'Bedworth Park', 'Boipatong', 'Bophelong',
            'Zamdela', 'Highbury', 'Roshnee', 'Grasmere', 'Midvaal', 'Daleside',
        )
        placeholders = ','.join('?' * len(VAAL_AREAS))
        qualify_where = (
            f"status='pending' "
            f"AND phone IS NOT NULL AND phone != '' "
            f"AND address IS NOT NULL AND address != '' "
            f"AND area IN ({placeholders})"
        )

        self._set_pipeline_btns(False)
        db_path = str(self.db.db_path)
        try:
            con = _sqlite3.connect(db_path, timeout=15)
            con.row_factory = _sqlite3.Row

            # Count qualifying records
            count = con.execute(
                f"SELECT COUNT(*) FROM businesses WHERE {qualify_where}", VAAL_AREAS
            ).fetchone()[0]

            if count == 0:
                self._log_pipeline("ℹ️  No pending businesses qualify for bulk publish.")
                return

            # Confirm with user (must run on main thread)
            confirmed = [False]
            def ask():
                import tkinter.messagebox as mb
                confirmed[0] = mb.askyesno(
                    "Bulk Publish",
                    f"This will publish {count:,} businesses.\n\n"
                    "Template descriptions will be generated for any without one.\n\n"
                    "Proceed?",
                    parent=self,
                )
            self.after(0, ask)
            # Wait for dialog (poll every 100ms, max 30s)
            import time
            for _ in range(300):
                time.sleep(0.1)
                if confirmed[0] is not False:
                    break
            if not confirmed[0]:
                self._log_pipeline("ℹ️  Bulk Publish cancelled.")
                return

            # Generate template descriptions for blank records
            self._log_pipeline("▶  Generating template descriptions…")
            blanks = con.execute(
                f"SELECT * FROM businesses "
                f"WHERE {qualify_where} "
                f"AND (description IS NULL OR description = '')",
                VAAL_AREAS,
            ).fetchall()
            desc_count = 0
            for row in blanks:
                desc = generate_template_description(dict(row))
                con.execute(
                    "UPDATE businesses SET description=? WHERE business_id=?",
                    (desc, row['business_id']),
                )
                desc_count += 1
            self._log_pipeline(f"✅  {desc_count:,} descriptions generated.")

            # Bulk publish
            self._log_pipeline("▶  Publishing businesses…")
            con.execute(
                f"UPDATE businesses SET status='published' WHERE {qualify_where}",
                VAAL_AREAS,
            )

            # Log to scrape_log
            con.execute(
                "INSERT INTO scrape_log "
                "(source_name, scrape_date, items_new, status) "
                "VALUES ('Bulk Publish', datetime('now'), ?, 'success')",
                (count,),
            )
            con.commit()
            con.close()
            self._log_pipeline(f"✅  Bulk Publish complete — {count:,} businesses published.")

        except Exception as exc:
            self._log_pipeline(f"❌  Bulk Publish error: {exc}")
        finally:
            self._set_pipeline_btns(True)
```

```javascript
const DAYS = ['Monday','Tuesday','Wednesday','Thursday','Friday','Saturday','Sunday'];
const DAYS_REGEX = new RegExp(`(${DAYS.join('|')})`, 'g');

function parseHours(raw) {
  if (!raw || typeof raw !== 'string') return null;
  // Raw format: "Monday8 am–5 pmTuesday8 am–5 pm..."
  const parts = raw.split(DAYS_REGEX).filter(Boolean);
  const result = [];
  for (let i = 0; i < parts.length - 1; i += 2) {
    if (DAYS.includes(parts[i])) {
      result.push({ day: parts[i], hours: parts[i + 1].trim() });
    }
  }
  return result.length ? result : null;
}
```
